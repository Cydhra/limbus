\begin{document}
    \paragraph{Expression Syntax}
    \begin{grammar}
        <expression> ::= <equality> .

        <equality> ::= <comparison> ( ( "!=" | "==" ) <comparison>)* .

        <comparison> ::= <addition> ( ( "<" | ">" | "<=" | ">=" ) <addition> )* .

        <addition> ::= <multiplication> ( ( "+" | "-" ) <multiplication> )* .

        <multiplication> ::= <unary> ( ( "/" | "*" ) <unary> )* .

        <unary> ::= ( "!" | "-" ) <unary> \alt <primary> .

        <primary> ::= <number>
%        \alt <string>
%        \alt "true"
%        \alt "false"
        \alt "(" <expression> ")" .
    \end{grammar}
